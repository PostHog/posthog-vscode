function switchTab(tab) {
    currentTab = tab;
    document.querySelectorAll('.nav-tab').forEach(t => t.classList.toggle('active', t.dataset.tab === tab));
    document.querySelectorAll('.section').forEach(s => s.classList.toggle('active', s.id === 'section-' + tab));
    document.getElementById('search').value = '';
    if (!loadedTabs.has(tab)) {
        loadedTabs.add(tab);
        if (tab === 'flags') send({ type: 'loadFlags' });
        else if (tab === 'experiments') send({ type: 'loadExperiments' });
        else if (tab === 'analytics') send({ type: 'loadInsights' });
    }
}

    **In script.ts** — Enhance `filterItems()` to show a "no results" message when all items are filtered out:

    Replace the existing `filterItems()` function with:
    ```javascript
    function filterItems() {
        const q = document.getElementById('search').value.toLowerCase();
        const list = document.getElementById(currentTab + '-list');
        if (!list) return;
        const selector = currentTab === 'analytics' ? '.insight-card' : '.item';
        const items = list.querySelectorAll(selector);
        let visibleCount = 0;
        items.forEach(item => {
            const text = item.textContent.toLowerCase();
            const show = text.includes(q);
            item.style.display = show ? '' : 'none';
            if (show) visibleCount++;
        });

        // Show/hide no-results message
        let noResults = list.querySelector('.no-results');
        if (visibleCount === 0 && q.length > 0 && items.length > 0) {
            if (!noResults) {
                noResults = document.createElement('div');
                noResults.className = 'no-results';
                noResults.textContent = 'No matching items';
                list.appendChild(noResults);
            }
            noResults.style.display = '';
        } else if (noResults) {
            noResults.style.display = 'none';
        }
    }
    ```

    Key changes from original:
    - Added null guard on `list` (`if (!list) return;`)
    - Added `visibleCount` tracking
    - Creates a `.no-results` div dynamically on first need (avoids HTML changes)
    - Shows it only when: zero visible items AND search query is non-empty AND there are items to filter
    - Hides it when items are visible or search is cleared
